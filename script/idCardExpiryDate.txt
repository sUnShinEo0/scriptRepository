import java.util.*;
import com.smy.framework.core.util.*;
import com.byxf.dcp.core.util.*;
String defValue = "1900-01-01";

if(StringUtils.isEmpty(custNo)){
    log("查询变量idCardExpiryDate传入参数为空!","debug");
    return defValue;
}
Map param = new HashMap();
param.put("cust_no",custNo);
List cstCustIdInfoList = (List)getDataSet(custIdInfo, param);

Map cstCustIdInfoMap = new HashMap();
if(CollectionUtil.isNotEmpty(cstCustIdInfoList)){
    cstCustIdInfoMap = (Map)cstCustIdInfoList.get(0);
}

if(CollectionUtil.isNotEmpty(cstCustIdInfoMap)){
    defValue = String.valueOf(cstCustIdInfoMap.get("ID_EXPIRE_DATE"));
    defValue=DateUtil.date2String(DateUtil.parseDate(defValue),DateUtil.YYYY_MM_DD_FORMAT)
}
return defValue;
